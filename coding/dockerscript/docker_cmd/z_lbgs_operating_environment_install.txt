#C/C++ and encryption
apt install libssl-dev -y

#check g++ install
g++ -v

# install cmake( >= 3.20)
apt remove cmake
wget https://github.com/Kitware/CMake/releases/download/v3.24.2/cmake-3.24.2-linux-x86_64.tar.gz
tar -xzvf cmake-3.24.2-linux-x86_64.tar.gz
rm -rf cmake-3.24.2-linux-x86_64.tar.gz
cd cmake-3.24.2-linux-x86_64/bin/
# add cmake command to env path
cd
vim .bashrc -> add line "PATH=$PATH:/root/download/cmake-3.24.2-linux-x86_64/bin"
:wq
cmake --version

# install docker(redis, postgresql)
apt-get remove docker \
               docker-engine \
               docker.io
apt-get update
sudo apt-get install \
     apt-transport-https \
     ca-certificates \
     curl \
     gnupg \
     lsb-release
curl -fsSL https://mirrors.aliyun.com/docker-ce/linux/debian/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
#official: curl -fsSL https://download.docker.com/linux/debian/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
echo \
"deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://mirrors.aliyun.com/docker-ce/linux/debian \
$(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
# official
#echo \
#"deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/debian \
#$(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
apt-get update
apt-get install docker-ce docker-ce-cli containerd.io

# install redis and postgresql
docker pull redis
docker pull postgres
